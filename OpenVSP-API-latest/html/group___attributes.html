for ( int i = 0; i < int( AttrIDs.size() ); i++ )
{
if ( AttrID == AttrIDs[i] )
{
AttrAdded = true;
}
}
DeleteAttribute( AttrID );
array < string > @NewAttrIDs = FindAllAttributes();
for ( int i = 0; i < int( NewAttrIDs.size() ); i++ )
{
if ( AttrID == NewAttrIDs[i] )
{
AttrDeleted = false;
}
}

{
Print( "Attribute successfully deleted" );
